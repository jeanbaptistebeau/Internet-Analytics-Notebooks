def read_edge_file(pathname):
    """Reads a graph from a csv file"""
    f = open(pathname, 'r')
    G = nx.Graph()
    for line in f:
        if line[0] == '#':
            continue
        line = f.readline().split(',')
        G.add_edge(int(line[0]) , int(line[1]))
    f.close()
    return G

def ccdf(graph):
    """ Computes the Complementary Cumulative Distribution 
    of the degree for the provided graph.
    Returns two numpy arrays, the first contains the degrees while
    the second contains the probabilities.
    """
    degrees = nx.degree(graph)
    deg_distr = Counter()
    for deg in degrees.values():
        deg_distr[deg] += 1
    cumulative = np.zeros(len(deg_distr))
    cumulative[0] = deg_distr[0]
    for i in range(1, len(cumulative)):
        cumulative[i] = deg_distr[i] + cumulative[i-1]
    return np.array(list(deg_distr.keys())), 1 - cumulative / len(graph)

def path_length_distribution(path_lengths):
    """Computes the distribution of the path lengths
    Returns a dictionary where the keys are the lengths
    and the values are the counts of paths having that length.
    """
    distribution = Counter()
    for dest in path_lengths.values():
        for dist in dest.values():
            distribution[dist] += 1
    return {k: distribution[k] / len(path_lengths) for k in distribution.keys()}

Exercise 2.6: Network identification¶

Clearly network1 is the subgraph of router comprising the Internet and network2 is the network of roads of New-York city. As seen in class one of the property of social network is to have a giant component and network1 has one. Another property about the connection between people, given by Duncan Watts, is that on average we are all connected by a path of length 6. Network1 has an average connection of 4.7 on his giant component (84% of the graph). Although newtork1 is not a graph of people but routers, we can imagine them having the same properties as routers connect humans throughout the web.
